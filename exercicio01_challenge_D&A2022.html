#Tratamento dos valores discrepantes por pontuao de intervalo interquartil(IQR)
cols = ['acoes_envolvidas', 'estoque_restante','valor_vendido','qtd_vendida']

Q1 = df_ofertas[cols].quantile(0.25)
Q3 = df_ofertas[cols].quantile(0.75)
IQR = Q3 - Q1

df_ofertas_out = df_ofertas[~((df_ofertas[cols] < (Q1 - 1.5 * IQR)) |(df_ofertas[cols] > (Q3 + 1.5 * IQR))).any(axis=1)]
